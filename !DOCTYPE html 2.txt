<script>
function search() {
  let query = document.getElementById("searchBox").value;
  let url = "https://api.duckduckgo.com/?q=" + query + "&format=json&pretty=1";

  fetch(url)
    .then(res => res.json())
    .then(data => {
      let resultDiv = document.getElementById("result");

      resultDiv.innerHTML = `
        <div class="box">
          <h3>${data.Heading}</h3>
          <p>${data.AbstractText}</p>
          <a href="${data.AbstractURL}" target="_blank">${data.AbstractURL}</a>
        </div>
      `;
    });
}
</script>
